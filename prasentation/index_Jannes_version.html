Konzept


\[\begin{gather} \text{Sateliten:} \\ \\ p_1 = (x_1; y_1) \\ p_2 = (x_2; y_2) \\ \\ p_r = (x_2 - x_1; y_2 - y_1) \\ \\ \end{gather} \]
\[\begin{gather} \text{Kreise:} \\ \\ x^2 + y^2 = r_1^2 \\ (x + x_r)^2 + (y+y_r)^2 = r_2^2 \\ \end{gather} \]
\[\begin{gather} \ (x^2 + 2xx_r + x_r^2) + (y^2 + 2yy_r + y_r^2) - (x^2 + y^2) = r_2^2 - r_1^2 \\ \ 2xx_r + x_r^2 + 2yy_r + y_r^2 = r_2^2 - r_1^2 &| - (r_2^2 - r_1^2) \\ \\ \ 2xx_r + x_r^2 + 2yy_r + y_r^2 -r_2^2 + r_1^2 = 0 &| - 2xx_r \\ \\ \ x_r^2 + 2yy_r + y_r^2 - r_2^2 + r_1^2 = -2xx_r &| :-2x_r \\ \\ \ \boxed{\frac{-(-r_2^2 + r_1^2 + 2yy_r +y_r^2 + x_r^2)}{2x_r} = x} \\ \end{gather}\]
\[\begin{gather} \ x = \frac{r_2^2 - r_1^2 \boxed{-2yy_r} - y_r^2 - x_r^2}{\boxed{2x_r}} = \frac{\boxed{-2yy_r}}{\boxed{2x_r}} + \underbrace{\frac{r_2^2 - r_1^2 - y_r^2 - x_r^2}{2x_r}}_{\sigma}\\ \ \sigma = \frac{r_2^2 - r_1^2 - y_r^2 - x_r^2}{2x_r} \\ \\ \ x = \frac{-2yy_r}{2x_r} + \sigma \\ \ \boxed{x = \frac{-yy_r}{x_r} + \sigma} \\ \end{gather} \]
\[\begin{gather} \ \boxed{x = \frac{-yy_r}{x_r} + \sigma} \quad \textrm{einsetzen in} \quad \boxed{x^2 + y^2 = r_1^2} \\ \\ \ \left(\frac{-yy_r}{x_r} + \sigma\right) ^2 + y^2 = r_1^2 \\ \ \frac{y_r^2}{x_r^2}y^2 -2 \frac{y_r\sigma}{x_r} y + \sigma^2 + y^2 = r_1^2 \\ \\ \ \boxed{(1 + \frac{y_r^2}{x_r^2}) y^2 + (- 2 \frac{y_r\sigma}{x_r})y + (\sigma^2 - r_1^2) = 0} \end{gather} \]
\[\begin{gather} \boxed{(1 + \frac{y_r^2}{x_r^2}) y^2 + (- 2 \frac{y_r\sigma}{x_r})y + (\sigma^2 - r_1^2) = 0} \\ a = 1 + \frac{y_r^2}{x_r^2} \\ b = - 2 \frac{y_r\sigma}{x_r} \\ c = \sigma^2 - r_1^2 \\ \\ a y^2 + by + c = 0 \\ \\ \end{gather}\]
\[\begin{gather} \boxed{\frac{-b \pm \sqrt{b^2 - 4ac}}{2a} = y_{1,2}} \\ \\ a = 1 + \frac{y_r^2}{x_r^2} \\ b = - 2 \frac{y_r\sigma}{x_r} \\ c = \sigma^2 - r_1^2 \\ \\ \boxed{\frac{2 \frac{y_r\sigma}{x_r^2} \pm \sqrt{4\frac{y_r^2\sigma^2}{x_r^2} - 4 \cdot (1 + \frac{y_r^2}{x_r^2}) \cdot (\sigma^2 - r_1^2)}}{2 + 2\frac{y_r^2}{x_r^2}} = y_{1,2}} \\ \\ \end{gather}\]
\[\begin{gather} \boxed{\frac{2 \frac{y_r \left( \frac{r_2^2 - r_1^2 - y_r^2 - x_r^2}{2x_r} \right)}{x_r^2} \pm \sqrt{4\frac{y_r^2 \left( \frac{r_2^2 - r_1^2 - y_r^2 - x_r^2}{2x_r} \right)^2}{x_r^2} - 4 \cdot (1 + \frac{y_r^2}{x_r^2}) \cdot (\left( \frac{r_2^2 - r_1^2 - y_r^2 - x_r^2}{2x_r} \right)^2 - r_1^2)}}{2 + 2\frac{y_r^2}{x_r^2}} = y_{1,2}} \\ \\ \end{gather}\] 
def testNewCalculation(c1, c2): 
	xr = c1[0][0] - c2[0][0]
	yr = c1[0][1] - c2[0][1]
	r1 = c1[1]
	r2 = c2[1]

	if (xr == 0 and yr == 0):
		return circlesSameSpot(c1, c2, False)
	
	if (math.sqrt(xr**2 + yr**2) > r1 + r2):  # Punkte zu weit entfernt 
		raise("Circles are too far apart") 

	d = (r2**2 - r1**2 - xr**2 - yr**2) / (2 * xr)
	a = 1 + (yr**2) / (xr**2)
	b = -2 * (d * yr) / xr
	c = d**2 - r1**2

	y = useABC(a, b, c, False)


	d = (r2**2 - r1**2 - xr**2 - yr**2) / (2 * yr)
	a = 1 + (xr**2) / (yr**2)
	b = -2 * (d * xr) / yr
	c = d**2 - r1**2

	x = useABC(a, b, c, False)

	return (x, y)

							function getIntersection(p1, p2, p3, d1, d2, d3, debug = false) {
								// distanz ausrechnen
								const r1Sq = d1 * d1; //6.25
								const r2Sq = d2 * d2; //12.96
								const r3Sq = d3 * d3; //17.64
								if (debug) console.log(r1Sq, r2Sq, r3Sq);
							
								// differenz zwischen zwei punkten
								const p2p1 = [p2[0] - p1[0], p2[1] - p1[1]]; // [3-0, 0-0] = [3, 0]
								const p3p1 = [p3[0] - p1[0], p3[1] - p1[1]]; // [0-0, 4-0] = [0, 4]
								if (debug) console.log(p2p1, p3p1);
							
								// punkt produkt (dot product fuck)
								const p2p1Sq = p2p1[0] * p2p1[0] + p2p1[1] * p2p1[1]; // [9+0]
								const p3p1Sq = p3p1[0] * p3p1[0] + p3p1[1] * p3p1[1]; // [0+16]
								if (debug) console.log(p2p1Sq, p3p1Sq)
							
								// koeffizienten und so lol
								const A = (r1Sq - r2Sq + p2p1Sq) / 2; // (6.25 - 12.96 + 9) / 2 = 1.145
								const B = (r1Sq - r3Sq + p3p1Sq) / 2; // (6.25 - 17.64 + 16) / 2 = 2.305
								if (debug) console.log(A, B)

								const y = (A * p3p1[0] - B * p2p1[0]) / (p2p1[1] * p3p1[0] - p3p1[1] * p2p1[0]); // (1.145 * 0 - 2.305 * 3) / (3 * 0 - 4 *)
								if (debug) console.log(x, y)

Fast Fourier (FFT)

  • ist ein mathematisches Verfahren, das in der Signalverarbeitung verwendet wird
  • ermöglicht die verschiedenen Frequenzanteile eines Signals zu analysieren

  • das Mikrofon nimmt alle Frequenzen auf, d.h. auch Hintergrundgeräusche
    => FFT zur Trennung von Stör- und Zielfrequenz

Hardware Grenzen

  • der Audio-Treiber startet nach einer unbestimmten Zeit sich neu
    • die Latenz des Mikrofons ändert sich davon abhängig

    => stark abweichende Resultate
